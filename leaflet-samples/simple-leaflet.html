    var map = L.map('map').setView([-3.6, 54.5], 5);

    var mvtSource = new L.TileLayer.MVTSource({
        url: "https://URL/{z}/{x}/{y}.pbf.pict",
        debug: true,

        style: function (feature) {
            var style = {};
            var type = feature.type;
            switch (type) {
                case 1: //'Point'
                    style.color = 'rgba(49,79,79,1)';
                    style.radius = 5;
                    style.selected = {
                        color: 'rgba(255,255,0,0.5)',
                        radius: 6
                    };
                    break;
                ...
            }
            return style;
        }

    });

    map.addLayer(mvtSource);
